function run(genFn) {
  return new Promise((resolve, reject) => {
    const gen = genFn;
    function step(nextFn) {
      let next;
      try {
        next = nextFn();
      } catch (err) {
        reject(err);
      }
      if (next.done) {
        resolve(next.value);
      }
      Promise.resolve(next.value).then(

      );
    }
    step(() => gen.next());
  });
}
