# 第 26 题： 千位分隔符

# 题目

给定一个整数，每千位加入 , 分隔符 示例：

1234567890 -> '1,234,567,890'

# 解答

数组分割

function format(num) {
  const [left, right] = `${num}`.split(".");
  const leftArr = left.split("");
  let pos = 0;
  let i = leftArr.length - 1;
  while (i >= 0) {
    pos += 1;
    if (pos % 3 === 1 && pos > 1 && i !== 0) {
      leftArr[i] = "," + leftArr[i];
    }
    i--;
  }
  return leftArr.join("") + "." + right;
}
