# 第 15 题：算法手写题

# 题目

给定两个数组，写一个方法来计算它们的交集。如：

给定 nums1 = [1, 2, 2, 1]，nums2 = [2, 2]，返回 [2, 2]。

# 解答

// 哈希表解法
function getCollection(arr1, arr2) {
  const result = [];
  const hashMap = {};
  arr1.forEach(item => {
    if (hashMap[item]) {
      hashMap[item]++;
    } else {
      hashMap[item] = 1;
    }
  });
  arr2.forEach(item => {
    if (hashMap[item] > 0) {
      result.push(item);
      hashMap[item]--;
    }
  });
  return result;
}

// 遍历
function getCollection(arr1, arr2) {
  const result = [];
  arr1.forEach(item1 => {
    arr2.forEach((item2, index) => {
      if (item1 === item2) {
        result.push(item2);
        arr2.splice(index, 1);
      }
    });
  });

  return result;
}
