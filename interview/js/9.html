function normalize(str) {
  let result;
  const arr = str.match(/\w+/g);
  while (arr.length) {
    let current = arr.pop();
    let temp = { value: current };
    if (result) {
      temp.children = result;
    }
    result = temp;
  }
  return result;
}

var s = "abc";
normalize(s); // {value: "abc"}

s = "[abc[bcd[def]]]";
normalize(s);
// {value: 'abc', children: {value: 'bcd', children: {value: 'def'}}}
