# 第 5 题：字符串数组去除重复的项，即['1','2','1','3'] —> ['1','2','3']

# 双重循环

function unique(array) {
  const newArray = [];
  let isRepeat;
  for (let i = 0; i < array.length; i++) {
    isRepeat = false;
    for (let j = 0; j < newArray.length; j++) {
      if (array[i] === newArray[j]) {
        isRepeat = true;
        break;
      }
    }
    if (!isRepeat) {
      newArray.push(array[i]);
    }
  }
  return newArray;
}

# Array.prototype.indexOf()

思路：如果索引不是第一个索引，说明是重复值。

  • 利用 Array.prototype.filter()过滤功能
  • Array.prototype.indexOf()返回的是第一个索引值
  • 只将数组中元素第一次出现的返回
  • 之后出现的将被过滤掉
function unique(array) {
  return array.filter((item, index) => {
    return array.indexOf(item) === index;
  });
}

# Array.prototype.sort()

思路：先对原数组进行排序，然后再进行元素比较。

function unique(array) {
  const newArray = [];
  array.sort();
  for (let i = 0; i < array.length; i++) {
    if (array[i] !== newArray[newArray.length - 1]) {
      newArray.push(array[i]);
    }
  }
  return newArray;
}

# Array.prototype.includes()

function unique(array) {
  const newArray = [];
  array.forEach(item => {
    if (!newArray.includes(item)) {
      newArray.push(item);
    }
  });
  return newArray;
}

# Map

function unique(array) {
  const map = new Map();
  return array.filter(item => {
    return !map.has(item) && map.set(item, true);
  });
}

# Set

function unique(array) {
  return Array.from(new Set(array));
