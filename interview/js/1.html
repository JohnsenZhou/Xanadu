# 第1题：什么是防抖和节流？有什么区别？如何实现？

# 防抖

触发高频事件后n秒内函数执行一次，如果n秒内高频事件再次触发，则重新计算时间

function debounce(func, delay) {
  let timer
  return function() {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      func.apply(this, arguments)
    }, delay)
  }
}

# 节流

高频事件触发，n秒内只执行一次，稀释执行频率

function throttle(func, delay) {
  let timer = null
  return function() {
    if (!timer) {
      timer = setTimeout(() => {
        func.apply(this, arguments)
        timer = null
      }, delay)
    }
  }
}
